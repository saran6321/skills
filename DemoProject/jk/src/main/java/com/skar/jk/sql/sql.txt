// create student table
CREATE TABLE student (
    roll_no INT PRIMARY KEY,
    name VARCHAR(255),  -- Assuming names can be up to 255 characters
    age INT,
    char CHAR,
    float FLOAT,
    boolean BOOLEAN,
    date DATE,
    datetime DATETIME
);

// create classroom table with relation to student
CREATE TABLE classroom (
    section VARCHAR(50),  -- You can adjust the maximum length as needed
    student_roll_no INT,
    FOREIGN KEY (student_roll_no) REFERENCES student(roll_no)
);

SELECT student.name, student.age                        // selects the column to retrive
FROM student                                            // from table
JOIN classroom
ON student.section = classroom.section                  // joins the tables with the primary key
WHERE classroom.section = 'YourDesiredSection';         // condition for retrival

SELECT student.name, student.age
FROM student
JOIN classroom
ON student.section = classroom.section
WHERE classroom.section = 'YourDesiredSection' AND student.age >= 18 ;
